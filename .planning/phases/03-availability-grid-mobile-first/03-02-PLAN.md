    **AvailabilityGrid** (`src/components/availability/availability-grid.tsx`):

    The main grid component. Props:
    ```typescript
    interface AvailabilityGridProps {
      dates: string[]        // ISO date strings "YYYY-MM-DD" — columns of the grid
      dayStart: number       // Hour 0-23 (e.g. 9 = 9am)
      dayEnd: number         // Hour 0-23 (e.g. 21 = 9pm)
    }
    ```

    **State and refs:**
    ```typescript
    const { selectedSlots, paintSlot, timezone } = useGridStore()
    const isDragging = useRef(false)
    const dragMode = useRef<'add' | 'remove'>('add')
    ```

    **Slot key generation:**
    Generate all slot times for the grid. For each date and each 30-min increment from dayStart to dayEnd:
    ```typescript
    import { fromZonedTime } from 'date-fns-tz'
    // fromZonedTime is the v3 name (NOT zonedTimeToUtc — that was removed in v3.0.0)

    function generateSlotKey(date: string, hour: number, minute: number, tz: string): string {
      const localStr = `${date}T${String(hour).padStart(2, '0')}:${String(minute).padStart(2, '0')}:00`
      return fromZonedTime(localStr, tz || 'UTC').toISOString()
    }
    ```

    If `timezone` is empty (not yet detected), use 'UTC' as fallback and re-render when timezone loads.

    **Time labels for TimeColumn:**
    ```typescript
    import { toZonedTime, format } from 'date-fns-tz'
    // toZonedTime is the v3 name (NOT utcToZonedTime — that was removed in v3.0.0)
    ```
    For each slot, convert the UTC slot key back to local time for display: `format(toZonedTime(new Date(slotKey), tz), 'h:mm a', { timeZone: tz })`.

    **Date column headers:**
    Format using `format(parseISO(date), 'EEE\nMMM d')` from date-fns — two lines, abbreviated day + month+day.

    **Pointer event handlers (CRITICAL — follow this pattern exactly):**

    ```typescript
    function getCellFromPoint(x: number, y: number): string | null {
      const el = document.elementFromPoint(x, y)
      return el?.closest('[data-slot-key]')?.getAttribute('data-slot-key') ?? null
    }

    function handlePointerDown(e: React.PointerEvent<HTMLDivElement>) {
      // Capture pointer on the container so pointermove fires even when finger moves outside
      e.currentTarget.setPointerCapture(e.pointerId)
      isDragging.current = true

      const slotKey = getCellFromPoint(e.clientX, e.clientY)
      if (!slotKey) return

      // First cell touched determines add vs remove for entire gesture
      dragMode.current = selectedSlots.has(slotKey) ? 'remove' : 'add'
      paintSlot(slotKey, dragMode.current)
    }

    function handlePointerMove(e: React.PointerEvent<HTMLDivElement>) {
      if (!isDragging.current) return
      const slotKey = getCellFromPoint(e.clientX, e.clientY)
      if (slotKey) paintSlot(slotKey, dragMode.current)
    }

    function handlePointerUp() {
      isDragging.current = false
    }
    ```

    **Grid DOM structure:**
    ```tsx
    <div className="overflow-x-auto">
      <div
        className="touch-none select-none"
        data-vaul-no-drag              // CRITICAL: prevents vaul drawer from intercepting grid drags
        style={{
          display: 'grid',
          gridTemplateColumns: `80px repeat(${dates.length}, minmax(64px, 1fr))`,
          minWidth: `${80 + dates.length * 64}px`,
        }}
        onPointerDown={handlePointerDown}
        onPointerMove={handlePointerMove}
        onPointerUp={handlePointerUp}
        onPointerCancel={handlePointerUp}
      >
        {/* Top-left corner cell (empty, sticky) */}
        <div className="sticky left-0 z-20 bg-[#FAF8F5]" />

        {/* Date column headers */}
        {dates.map(date => (
          <div key={date} className="text-center text-xs font-medium text-[#6B6158] py-2 border-b border-[#E5DDD4] whitespace-pre-line">
            {format(parseISO(date), 'EEE\nMMM d')}
          </div>
        ))}

        {/* Rows: one per 30-min slot */}
        {slotRows.map(({ slotKey, label }) => (
          <React.Fragment key={slotKey}>
            {/* Sticky time label */}
            <div className="sticky left-0 z-10 bg-[#FAF8F5] text-xs text-[#6B6158] min-h-[44px] flex items-start pt-1 px-2 border-b border-[#E5DDD4]">
              {label}
            </div>
            {/* Cells across all date columns */}
            {dates.map(date => {
              const cellKey = generateSlotKey(date, slotHour, slotMinute, timezone)
              return (
                <GridCell
                  key={`${date}-${slotKey}`}
                  slotKey={cellKey}
                  isSelected={selectedSlots.has(cellKey)}
                />
              )
            })}
          </React.Fragment>
        ))}
      </div>
    </div>
    ```

    Mark this file `'use client'`.

    **Anti-patterns to avoid:**
    - DO NOT use React state during drag — isDragging and dragMode are refs, paint calls go through Zustand
    - DO NOT set pointer capture on individual GridCell elements — it MUST be on the container
    - DO NOT forget `data-vaul-no-drag` on the grid container — without it, vertical drags dismiss the vaul drawer
    - DO NOT forget `touch-none` (touch-action: none) — without it, iOS Safari intercepts touch events for scroll
